# Boxplot of AQI values by Status
plt.figure(figsize=(12, 6))
sns.boxplot(x="Status", y="AQI Value",hue="Status", data=df, order=df["Status"].value_counts().index,palette="coolwarm")
plt.title("AQI Distribution by Air Quality Status")
plt.xlabel("Air Quality Status")
plt.ylabel("AQI Value")
plt.xticks(rotation=45)
plt.show()
In[7]:
# Bar plot of average AQI per country (top 10 countries)
top_countries = df.groupby("Country")["AQI Value"].mean().nlargest(10).index
plt.figure(figsize=(12, 6))
sns.barplot(x="Country", y="AQI Value",hue="Country", data=df[df["Country"].isin(top_countries)],palette="magma")
plt.title("Top 10 Countries with Highest Average AQI")
plt.xlabel("Country")
plt.ylabel("Average AQI Value")
plt.xticks(rotation=45)
plt.show()
